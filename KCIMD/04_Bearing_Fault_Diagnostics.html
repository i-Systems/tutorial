# Show STFT of first data for each class  
# Use fs = 12000
#     windowsize = 110

data = [normal, inner, outer, ball]

Fs = 12000       
windowsize = 110
window = np.hanning(windowsize)
nfft = windowsize
noverlap = windowsize/2

plt.figure(figsize = (12, 12))

for i in range(4):
    f, t, Sxx = spectrogram(data[i][0], 
                            Fs, 
                            window = window, 
                            noverlap = noverlap, 
                            nfft = nfft, 
                            scaling = 'spectrum', 
                            mode = 'psd')

    plt.subplot(2, 2, i+1)
    plt.pcolormesh(t, f, Sxx)
    plt.title('Classes : {}'.format(classes[i]), fontsize = 15)
    plt.ylabel('Frequency [Hz]', fontsize = 12)
    plt.xlabel('Time [sec]', fontsize = 12)

plt.show()

3. CNN Structure

In[10]:
# Define model
# Use 3 layers(convolution + pooling)
#     1 GAP
#     1 Dense layer

model = tf.keras.models.Sequential([
    tf.keras.layers.Conv2D(32, (3,3), activation = 'relu', padding = 'SAME', input_shape = (56, 56, 1)),
    tf.keras.layers.MaxPool2D((2,2)),
    
    tf.keras.layers.Conv2D(64, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.MaxPool2D((2,2)),
    
    tf.keras.layers.Conv2D(64, (3,3), activation = 'relu', padding = 'SAME'),
    tf.keras.layers.MaxPool2D((2,2)),
    
    tf.keras.layers.GlobalAveragePooling2D(),
    tf.keras.layers.Dense(4, activation = 'softmax', use_bias = False)
])

model.summary()

4. Visualization of Classification Results using CAM

In[14]:
# get max pooling layer and fully connected layer 
conv_layer = model.get_layer(index = 5)
fc_layer = model.get_weights()[6]

# Class activation map 
activation_map = tf.matmul(conv_layer.output, fc_layer)
CAM_model = tf.keras.Model(inputs = model.inputs, outputs = activation_map)
In[15]:
# CAM for random test image 
test_img = test_imgs[np.random.choice(test_imgs.shape[0], 1)]

pred = np.argmax(model.predict(test_img), axis = 1)

CAM = CAM_model.predict(test_img)
attention = CAM[:,:,:,pred]
attention = np.abs(np.reshape(attention,(7,7)))

test_x = test_img.reshape(56,56)
large_attention = cv2.resize(attention, (56, 56), interpolation = cv2.INTER_CUBIC)

plt.figure(figsize = (10,10))
plt.subplot(2,2,1)
plt.imshow(test_x, origin='lower')
plt.title('Input Image')
plt.axis('off')
plt.subplot(2,2,2)
plt.title('CAM')
plt.imshow(large_attention, 'jet', origin='lower', alpha = 0.5)
plt.axis('off')
plt.subplot(2,1,2)
plt.imshow(test_x)
plt.imshow(large_attention, 'jet', origin='lower', alpha = 0.5)
plt.title('Input Image + CAM')
plt.axis('off')
plt.show()
